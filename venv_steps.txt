# HOW TO: Virtual environments (from https://youtu.be/Kg1Yvry_Ydk)




# this creates a new folder inside the root project directory called venv
python3 -m venv venv



# this will activate the virtual environment
source venv/bin/activate
# and the shell will then display: (venv) 


# to show the current libraries/packages installed in current virtual env type
pip3 list 


# type 
pip3 install -r requirements.txt
# to installed the required packages needed to
# make the project work.
# it also includes the versions installed on my
# machine at the time of the testing of the project
# requirements-short.txt is a quicker
# version and should also work fine, but does not include versions



# to quit the virtual environment type
deactivate



# to delete the virtual env
rm -rf venv/
# like any other directory
